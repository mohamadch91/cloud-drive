  /**
   * convert string to persian numbers
   * @param {string} str input string
   * @returns persian converted string
   */
  stringconvertor = (str) => {
    let newstr="";
    for(let i=0;i<str.length;i++){
      // console.log(str)
      if(str[i]==="1"){
        newstr+="١";
      }
      else if(str[i]==="2"){
        newstr+="٢";
      }
      else if(str[i]==="3"){
        newstr+="٣";
      }
      else  if(str[i]==="4"){
        newstr+="۴";
      }
      else  if(str[i]==="5"){
        newstr+="۵";
      }
      else  if(str[i]==="6"){
        newstr+="۶";
      }
      else if(str[i]==="7"){
        newstr+="٧";
      }
      else  if(str[i]==="8"){
        newstr+="٨";
      }
      else   if(str[i]==="9"){
        newstr+="٩";
      }
      else if(str[i]==="."){
        newstr+="."
      }
      else  if(str[i]==="0"){
        newstr+="٠";
      }
      else{
        newstr+=str[i];
      }
     
    }
    return newstr;
  }

    if (file_size >= 1000000) {
      x = file_size / 1000000;
      x = x.toFixed(2);
      x=this.stringconvertor(x);
      arr[0]=x;
      arr[1]=" مگابایت"

      x = file_size / 1000;
      x = x.toFixed(2);
      arr[0]=x;
      arr[1]=" کیلوبایت"
    } else if (file_size > 1000000000) {
      x = file_size / 1000000000;
      x = x.toFixed(2);
      arr[0]=x;
      arr[1]="گیگابایت"
    } else {
      x = file_size;
      x = x.toFixed(2);
      arr[0]=x;
      arr[1]=" بایت"
    }
    return arr;
  }
